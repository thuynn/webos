Test_SNF_WOS_12_ManageApplicationIntoCategory
open /portal/public/classic/
click link=Sign in
type username root
type password gtn
clickAndWait //div[@id='UIPortalLoginFormAction']/div/div/div/a
clickAndWait link=Desktop
echo -- Open Application Registry portlet --
mouseMoveAt //img[@src='/eXoResources/skin/sharedImages/Icon80x80/DefaultPortlet.png']
click //img[@src='/eXoResources/skin/sharedImages/Icon80x80/DefaultPortlet.png']
verifyTextPresent Application Registry
clickAt //div[@id='UIPageDesktop']//div[1]//div[@title='Maximize']
echo -- Add new category--
waitForElementPresent css=div.IconControl.AddCategoryIcon
click css=div.IconControl.AddCategoryIcon
waitForElementPresent name
type name Test_SNF_WOS_12
type displayName Test_SNF_WOS_12
type description Test_SNF_WOS_12
echo -- Choose Permission Setting tab--
waitForElementPresent //div[text()='Permission Setting']
click //div[text()='Permission Setting']
waitForElementPresent link=Add Permission
click link=Add Permission
waitForElementPresent //a[@title='Organization']
click //a[@title='Organization']
waitForElementPresent link=exact:*
click link=exact:*
waitForElementPresent link=Save
click link=Save
waitForTextPresent Test_SNF_WOS_12
verifyTextPresent Test_SNF_WOS_12
echo -- Add application into category--
waitForElementPresent //a[@title="Add application to category"]
click //a[@title="Add application to category"]
waitForElementPresent //input[@name='application' and @value='5']
click //input[@name='application' and @value='5']
waitForElementPresent displayName Test_SNF_WOS_12
type displayName Test_SNF_WOS_12
waitForElementPresent link=Add
click link=Add
waitForTextPresent Test_SNF_WOS_12
verifyTextPresent Test_SNF_WOS_12
echo -- Edit application into category--
waitForElementPresent //div[contains(@onclick,'EditApplication') and @title='Edit Application']
click //div[contains(@onclick,'EditApplication') and @title='Edit Application']
waitForElementPresent displayName
type displayName Test_SNF_WOS_12_edit
waitForElementPresent link=Save
click link=Save
waitForTextPresent Test_SNF_WOS_12_edit
verifyTextPresent Test_SNF_WOS_12_edit
echo -- Delete application into category--
waitForElementPresent //a[contains(@onclick,'RemoveApplication') and @title='Delete application']
click //a[contains(@onclick,'RemoveApplication') and @title='Delete application']
waitForTextNotPresent Test_SNF_WOS_12_edit
verifyTextNotPresent Test_SNF_WOS_12_edit
echo -- Delete category--
waitForElementPresent //a[contains(@onclick,'Test_SNF_WOS_12') and @title='Delete category']
click //a[contains(@onclick,'Test_SNF_WOS_12') and @title='Delete category']
waitForConfirmationPresent
assertConfirmation Are you sure to delete this category and all applications on it?
waitForTextNotPresent Test_SNF_WOS_12
verifyTextNotPresent Test_SNF_WOS_12
echo -- Quit application registry from dockbar--
clickAt //div[@id='UIPageDesktop']//div[1]//div[@title='Minimize']
mouseDownRightAt //img[contains(@src,'DefaultPortlet.png') and contains(@onmousedown,'UIRightClickPopupMenu')] 3,4
waitForElementPresent link=Quit
click link=Quit
waitForElementPresent link=Sign out
clickAndWait link=Sign out
