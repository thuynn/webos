Test_SNF_WOS_10_ImportApplication
open /portal/public/classic/
click link=Sign in
type username root
type password gtn
clickAndWait //div[@id='UIPortalLoginFormAction']/div/div/div/a
clickAndWait link=Desktop
echo -- Open Application Registry portlet --
mouseMoveAt //img[@src='/eXoResources/skin/sharedImages/Icon80x80/DefaultPortlet.png']
click //img[@src='/eXoResources/skin/sharedImages/Icon80x80/DefaultPortlet.png']
verifyTextPresent Application Registry
clickAt //div[@id='UIPageDesktop']//div[1]//div[@title='Maximize']
waitForText //div[@id='UIApplicationOrganizer']/div[2]/div[1]/div/div/div[2]/div/div[1]/div[1]/div/div/div/a[1] Administration
verifyText //div[@id='UIApplicationOrganizer']/div[2]/div[1]/div/div/div[2]/div/div[1]/div[1]/div/div/div/a[1] Administration
waitForText //div[@id='UIApplicationOrganizer']/div[2]/div[1]/div/div/div[2]/div/div[2]/div/div/div/div/a Dashboard
verifyText //div[@id='UIApplicationOrganizer']/div[2]/div[1]/div/div/div[2]/div/div[2]/div/div/div/div/a Dashboard
waitForText link=Gadgets Gadgets
verifyText link=Gadgets Gadgets
waitForText link=web web
verifyText link=web web
echo -- Click Import Application --
waitForElementPresent //div[text()='Import Applications']
click //div[text()='Import Applications']
assertConfirmation This action will automatically create categories and import all the gadgets and portlets on it.
echo -- Show existing portlets and category are imported successfully--
waitForTextPresent Administration
verifyTextPresent Administration
waitForTextPresent Dashboard
verifyTextPresent Dashboard
waitForText link=Gadgets Gadgets
verifyText link=Gadgets Gadgets
waitForText link=Integration Integration
verifyText link=Integration Integration
waitForText link=Navigation Navigation
verifyText link=Navigation Navigation
verifyText link=System System
verifyText link=System System
waitForText link=web web
verifyText link=web web
echo -- Delete some category after imported--
waitForElementPresent link=Integration
click link=Integration
waitForElementPresent //a[contains(@onclick,'Integration') and @title='Delete category']
click //a[contains(@onclick,'Integration') and @title='Delete category']
waitForConfirmationPresent
assertConfirmation Are you sure to delete this category and all applications on it?
waitForElementPresent link=Navigation
click link=Navigation
waitForElementPresent //a[contains(@onclick,'Navigation') and @title='Delete category']
click //a[contains(@onclick,'Navigation') and @title='Delete category']
assertConfirmation Are you sure to delete this category and all applications on it?
waitForElementPresent link=System
click link=System
waitForElementPresent //a[contains(@onclick,'System') and @title='Delete category']
click //a[contains(@onclick,'System') and @title='Delete category']
waitForConfirmationPresent
assertConfirmation Are you sure to delete this category and all applications on it?
echo -- Quit application registry from dockbar--
clickAt //div[@id='UIPageDesktop']//div[1]//div[@title='Minimize']
mouseDownRightAt //img[contains(@src,'DefaultPortlet.png') and contains(@onmousedown,'UIRightClickPopupMenu')] 3,4
waitForElementPresent link=Quit
click link=Quit
waitForElementPresent link=Sign out
clickAndWait link=Sign out
